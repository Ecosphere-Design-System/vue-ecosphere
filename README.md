### Usage

Install

```sh
npm install vue-ecosphere
```

Import the plugin in `main.js`

```js
import { createApp } from "vue";
import Ecosphere from "vue-ecosphere";

const app = createApp(App);
app.use(Ecosphere);
app.mount("#app");
```
